a-color-picker

A color picker for web app

Default

AColorPicker.createPicker(el);

Small

AColorPicker.createPicker({
	attachTo: el,
	color: '#DD6E42',
	showHSL: false });

With opacity

AColorPicker.createPicker({
    attachTo: el,
    color: 'rgba(121, 69, 169, .6)',
    showAlpha: true });

With palette

AColorPicker.createPicker({ 
    attachTo: el,
    color: '#03a9f4', 
    palette: 
        AColorPicker.PALETTE_MATERIAL_CHROME });

With palette editable

AColorPicker.createPicker({
    attachTo: el,
    color: '#03a9f4',
    palette: 
        AColorPicker.PALETTE_MATERIAL_CHROME,
    paletteEditable: true });
